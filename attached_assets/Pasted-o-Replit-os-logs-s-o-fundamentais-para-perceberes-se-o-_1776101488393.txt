Como criar um Log Detalhado (Exemplo em Node.js)Se o teu sistema não estiver a imprimir detalhes suficientes, podes adicionar uma função de log que regista tudo num ficheiro chamado atividades.log. Assim, podes abrir esse ficheiro a qualquer momento para ver o histórico.O Código de LogJavaScriptconst fs = require('fs');

function registrarLog(mensagem) {
    const timestamp = new Date().toLocaleString('pt-BR');
    const linhaLog = `[${timestamp}] ${mensagem}\n`;
    
    // Escreve no console para veres agora
    console.log(linhaLog);
    
    // Guarda num ficheiro para veres depois
    fs.appendFileSync('prospeccao.log', linhaLog);
}

// EXEMPLO DE COMO USAR NO TEU SISTEMA:
registrarLog("Iniciando busca por: 'Transportadoras Guarulhos'");

// Se encontrar um lead:
registrarLog("✅ Lead encontrado: Transportes ABC - Email: contato@abc.com.br");

// Se falhar:
registrarLog("⚠️ Aviso: O site transportesxyz.com.br não permitiu a leitura de e-mails.");
