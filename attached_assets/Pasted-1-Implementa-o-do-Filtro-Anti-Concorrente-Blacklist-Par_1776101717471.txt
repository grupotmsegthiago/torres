1. Implementação do Filtro Anti-Concorrente (Blacklist)
Para garantir que a Torres Vigilância não envie apresentações para outras empresas de escolta, precisamos de injetar uma lógica de "limpeza" imediata após a captura do site.

O que adicionar no teu código de filtragem:
JavaScript
// Lista de termos proibidos (Concorrentes)
const BLACKLIST_SEGURANÇA = [
    "escolta armada", "vigilância", "segurança patrimonial", 
    "seguranca privada", "monitoramento eletrônico", "vms",
    "portaria remota", "segurança eletrônica", "vigilante"
];

// Função de validação rápida
function validarLead(conteudoSite) {
    const texto = conteudoSite.toLowerCase();
    
    // Se encontrar qualquer termo de segurança, retorna FALSE (não é lead)
    const ehConcorrente = BLACKLIST_SEGURANÇA.some(termo => texto.includes(termo));
    
    if (ehConcorrente) {
        return { valido: false, motivo: "Concorrente detectado" };
    }
    return { valido: true };
}

Usa: "logística São Paulo -vigilância -escolta -segurança"
